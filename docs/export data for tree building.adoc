= SQL for export

```sql
SELECT
    id AS key,
    CASE WHEN gender = 1 THEN
        'M'
    ELSE
        'F'
    END AS s,
    surname AS surn,
    COALESCE(last_surname, '') AS marn,
    name AS firn,
    patronymic AS secn,
    COALESCE(father_id::text, '') AS f,
    COALESCE(mother_id::text, '') AS m,
    birthday AS bday,
    COALESCE(deathday::text, '') AS dday,
    surname || ' ' || name || ' ' || person.patronymic AS n
FROM person
```
